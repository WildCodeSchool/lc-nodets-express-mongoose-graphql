CREATE TABLE "main"."Wilder" (
    id SERIAL PRIMARY KEY NOT NULL,
    name VARCHAR(255),
    city VARCHAR(255)
);

CREATE TABLE "main"."Skills" (
    id SERIAL PRIMARY KEY NOT NULL,
    votes VARCHAR(255),
    title VARCHAR(255),
    "wilderId" INTEGER NOT NULL,
    FOREIGN KEY ("wilderId") REFERENCES "Wilder"(id)
);
